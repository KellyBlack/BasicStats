\begin{frame}
  \frametitle{Conditional Probability}

  \begin{definition}
    The probability that A occurs given that I know that B occured is
    defined to be 
    \begin{eqnarray*}
      p(A|B) & = & \frac{p(A \mathrm{~and~} B)}{p(B)}, \\
      \mathrm{or~} 
      p(B) p(A|B)  &  =  & p(A \mathrm{~and~} B).
    \end{eqnarray*}
  \end{definition}

\end{frame}

\section{Compliment}

\begin{frame}{Compliment}

  \begin{definition}
    Suppose $A$ is an event. Then all of the events not in A are denoted
    by $\bar{A}$. 

    Other notations:
    \begin{eqnarray*}
      \bar{A},~A',~A^c
    \end{eqnarray*}
  \end{definition}


\end{frame}


\begin{frame}{Probabilities of Compliments}
  
  \begin{eqnarray*}
    1 & = & p(A) + p(\bar{A}), \\
    p(A) & = & 1 - p(\bar{A}).
  \end{eqnarray*}

\end{frame}
