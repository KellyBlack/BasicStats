\subsection{Cumulative Frequency}


\begin{frame}
  \frametitle{Cumulative Frequency}

  \begin{definition}
    The Cumulative Frequency is the frequncy of occurances for each
    result \textbf{less} than the given value.
  \end{definition}

\end{frame}
