\subsection{Events}

\begin{frame}
  \frametitle{Events}

  \begin{definition}
    An event is a possible outcome from an experiment.
  \end{definition}

\end{frame}


\begin{frame}
  \frametitle{Sample Space}

  \begin{definition}
    The sample space is the collection of all possible events.
  \end{definition}

\end{frame}


\begin{frame}
  \frametitle{Sample Space}

  There are two ways to visualize the sample space:
  \begin{itemize}
  \item Venn Diagram
  \item Tree Diagram
  \end{itemize}

\end{frame}

\subsection{Properties of Probabilities}

\begin{frame}{Properties of Probabilities}

  Suppose that A is an event in the sample space, then
  \begin{eqnarray*}
    \begin{array}{rcccl}
      0 & \leq & P(A) & \leq & 1
    \end{array}
  \end{eqnarray*}
  
\end{frame}
