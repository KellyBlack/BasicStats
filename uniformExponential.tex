\subsection{Clicker Quiz}


\iftoggle{clicker}{%
  \begin{frame}
    \frametitle{Clicker Quiz}

    A random variable, $X$, has a probability density function given by
    \begin{eqnarray*}
      f(x) & = & \left\{
        \begin{array}{l@{\hspace{2em}}l}
          \frac{1}{8}, & 1 \leq x \leq 9, \\
          0 & \mathrm{otherwise.}
        \end{array}
      \right.
    \end{eqnarray*}
    What is the probability that $X$ is less than three?

    \vfill
    
    \begin{tabular}{l@{\hspace{3em}}l@{\hspace{3em}}l@{\hspace{3em}}l}
      A: $\frac{2}{8}$ & B: $\frac{3}{8}$ & C: $\frac{4}{8}$  & D: $\frac{5}{8}$
    \end{tabular}

  \end{frame}

\subsection{Uniform Distribution}


 \begin{frame}
 \frametitle{Example}

 A random variable, $X$, is uniformly distributed between one and
 five. What is the probability that it is between 2.5 and 3.5?

 \end{frame}


 \begin{frame}
 \frametitle{Example}

 You pick a resistor at random from a batch. The resistance in the
 resistors is uniformly distributed between $2.8\times 10^6$ and
 $3.2\times 10^6$. What is the probability that the resistance is less
 than $2.9\times 10^6$?

\end{frame}

\subsection{Exponential Distribution}


\iftoggle{clicker}{%
  \begin{frame}
    \frametitle{Clicker Quiz}

    The mean time to failure for a disk drive is 250,000
    hours. Assuming that the mean time before failure follows an
    exponential distribution what is the value of $\lambda$?

    \vfill
    
    \begin{tabular}{l@{\hspace{3em}}l@{\hspace{3em}}l@{\hspace{3em}}l}
      A: $\frac{1}{250,000}$ & B: $\sqrt{\frac{1}{250,000}}$ & C: $250,000$  & D: $\sqrt{250,000}$
    \end{tabular}

    \vfill
    \vfill
    \vfill

  \end{frame}
}



\begin{frame}{Example}

  The mean time to failure for a disk drive is 250,000 hours. What is
  the probability that a disk drive chosen at random will not last
  past 200,000 hours? (Assume an exponential distribution.)
  
\end{frame}

\begin{frame}{Example}

  The mean time to failure for a disk drive is 250,000 hours. What is
  the probability that a disk drive chosen at random will not last
  past an additional 200,000 hours given that it has been running
  successfully for the last 100,000 hours? (Assume an exponential
  distribution.)
  
\end{frame}
